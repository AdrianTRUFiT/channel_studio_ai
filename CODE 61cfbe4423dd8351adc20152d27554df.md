import os

from datetime import datetime

from crewai import Agent, Task, Crew, Process, LLM, tool

from crewai_tools import FileReadTool, DirectoryReadTool, FileWriteTool

from crewai.memory import Memory

# ====================== CONFIG ======================

BASE_DIR = "D:\\Studio"

os.makedirs(f"{BASE_DIR}\\Scripts", exist_ok=True)

os.makedirs(f"{BASE_DIR}\\Output", exist_ok=True)

os.makedirs(f"{BASE_DIR}\\Products", exist_ok=True)

os.makedirs(f"{BASE_DIR}\\Journal", exist_ok=True)

local_llm = LLM(

model="ollama/llama3.1:8b",  # Change to llama3.2 or qwen2.5:14b if available

base_url="http://localhost:11434",

temperature=0.65

)

# ====================== MEMORY ======================

studio_memory = Memory(

llm=local_llm,

embedder={

"provider": "ollama",

"config": {"model_name": "mxbai-embed-large"}

},

recency_weight=0.4,

semantic_weight=0.4,

importance_weight=0.2,

recency_half_life_days=45,

consolidation_threshold=0.82,

)

# ====================== CUSTOM TOOLS ======================

@tool("Save_Script")

def save_script(content: str, filename: str) -> str:

"""Saves script to Scripts folder"""

path = f"{BASE_DIR}\\Scripts\\{filename}.txt"

with open(path, "w", encoding="utf-8") as f:

f.write(content)

return f"✅ Script saved: {path}"

@tool("Save_Metadata")

def save_metadata(title: str, description: str, tags: str) -> str:

"""Saves YouTube metadata"""

path = f"{BASE_DIR}\\Output\\metadata_{title.replace(' ', '_')}.txt"

content = f"Title: {title}\n\nDescription:\n{description}\n\nTags:\n{tags}"

with open(path, "w", encoding="utf-8") as f:

f.write(content)

return f"✅ Metadata saved: {path}"

@tool("Generate_Digital_Product")

def generate_digital_product(script: str, product_type: str) -> str:

"""Creates PDF-ready content for digital products"""

product = f"# {product_type} from Trusting AI\n\n{script}\n\n---\nCreated by Channel Studio AI"

path = f"{BASE_DIR}\\Products\\{product_type.replace(' ', '_')}.md"

with open(path, "w", encoding="utf-8") as f:

f.write(product)

return f"✅ Digital product created: {path}"

# ====================== AGENTS ======================

project_manager = Agent(

role="Studio Project Manager",

goal="Orchestrate the full video production + revenue pipeline. Ensure kid-friendly quality and maximum revenue potential.",

backstory="Organized, encouraging leader who helps 10-year-olds run a professional faceless YouTube channel focused on Trusting AI and AI Governance.",

llm=local_llm,

allow_delegation=True,

verbose=True,

memory=True

)

researcher = Agent(

role="Trusting AI Book Researcher & Script Writer",

goal="Extract high-quality educational content from the Trusting AI book and create engaging 250-400 word scripts.",

backstory="Expert at simplifying AI Governance, Authorization, Trust, and Ethics for young audiences.",

llm=local_llm,

tools=[FileReadTool(), DirectoryReadTool(directory=f"{BASE_DIR}\\Instructions")],

allow_delegation=False,

verbose=True

)

visuals_agent = Agent(

role="Visuals & Thumbnail Designer",

goal="Create strong image generation prompts for Grok Imagine / Canva and thumbnail concepts.",

backstory="Creative visual expert who makes faceless videos engaging.",

llm=local_llm,

allow_delegation=False,

verbose=True

)

capcut_agent = Agent(

role="CapCut Automation Specialist",

goal="Deliver precise, copy-paste CapCut automation steps including Auto Captions, music balance, and export.",

backstory="Master of CapCut for fast, high-quality educational faceless videos.",

llm=local_llm,

allow_delegation=False,

verbose=True

)

revenue_agent = Agent(

role="SEO & Revenue Optimizer",

goal="Create high-performing YouTube metadata, affiliate recommendations, and digital product ideas that drive consistent cash flow.",

backstory="Smart marketer who balances education with honest monetization (affiliates, courses, products).",

llm=local_llm,

tools=[save_metadata, generate_digital_product],

allow_delegation=False,

verbose=True

)

# ====================== TASKS ======================

task_research = Task(

description="Read the current request or relevant 'Trusting AI' book section. Produce a complete 250-400 word script for a 2-5 minute video. Include 1-2 natural affiliate suggestions and visual cues.",

agent=researcher,

expected_output="Full script with affiliate ideas and visual notes.",

tools=[FileReadTool()]

)

task_visuals = Task(

description="Based on the script, create 8-12 detailed image prompts and one high-click thumbnail prompt suitable for Grok Imagine or Canva.",

agent=visuals_agent,

expected_output="Numbered list of image prompts + thumbnail prompt."

)

task_capcut = Task(

description="Provide exact numbered CapCut Desktop automation steps: import voiceover, AutoCut, Auto Captions (with optimal settings: English, highlight keywords ON, filler words ON), background music (15-30% volume), transitions, and 1080p export.",

agent=capcut_agent,

expected_output="Detailed step-by-step CapCut instructions ready for the kids."

)

task_revenue = Task(

description="Create optimized YouTube Title, full Description (with affiliate disclosure at top + timestamps + product link), 10-15 tags. Also generate one sellable digital product idea (cheat sheet, workbook, or mini-course module).",

agent=revenue_agent,

expected_output="Complete metadata package + digital product content.",

tools=[save_metadata, generate_digital_product]

)

task_final = Task(

description="Review all outputs from the team. Create a clear kid-friendly checklist, summarize revenue opportunities, and give final recommendations.",

agent=project_manager,

expected_output="Final approval checklist + next actions."

)

# ====================== CREW ======================

studio_crew = Crew(

agents=[researcher, visuals_agent, capcut_agent, revenue_agent, project_manager],

tasks=[task_research, task_visuals, task_capcut, task_revenue, task_final],

process=Process.hierarchical,

manager_agent=project_manager,

memory=studio_memory,

verbose=True,

cache=True

)

# ====================== MAIN EXECUTION ======================

if __name__ == "__main__":

print("🚀 Channel Studio AI Crew Starting...\n")

topic = input("Enter video topic or book chapter (e.g., 'AI Authorization' or 'Trusting AI Chapter 5'): ").strip()

if not topic:

topic = "AI Governance Authorization"

result = studio_crew.kickoff(inputs={

"topic": topic,

"book": "Trusting AI",

"date": datetime.now().strftime("%Y-%m-%d"),

"base_dir": BASE_DIR

})

output_path = f"{BASE_DIR}\\Output\\full_run_{topic.replace(' ', '_')}_{datetime.now().strftime('%H%M')}.txt"

with open(output_path, "w", encoding="utf-8") as f:

f.write(f"Channel Studio AI - Run on {datetime.now()}\n")

f.write(f"Topic: {topic}\n\n")

f.write(str(result))

print("\n" + "="*80)

print("✅ CHANNEL STUDIO AI CREW COMPLETED SUCCESSFULLY!")

print("="*80)

print(f"📁 Full output saved to: {output_path}")

print(f"📁 Scripts saved in: {BASE_DIR}\\Scripts")

print(f"📁 Digital products in: {BASE_DIR}\\Products")

print("\nNext steps for the kids:")

print("1. Review the script and metadata")

print("2. Generate voiceover in CapCut or ElevenLabs")

print("3. Follow CapCut instructions")

print("4. Upload to YouTube with metadata")

print("5. Sell the generated digital product on Gumroad")

print("\nThe system is now ready to scale revenue! 💰")
